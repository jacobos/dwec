function library(name, num_books) {
            this.name = name;
            this["# num lib"] = 0,
            this.addBook = function (seccion, title, autor) {
                this["# num lib"]++;
                return this[seccion]["libro_" + this["# num lib"]] = {
                    title: title,
                    author: autor
                };

            };
            this.addSeccion = function (seccion) {
                if (Array.isArray(seccion)) {
                    var len = seccion.length;
                    for (var i = 0; i < len; i = i + 1) {
                        this[seccion[i]] = {};
                    }
                } else {
                    this[seccion] = {};
                }
            }
        };
        library.prototype.showBook = function (section, libro) {
            return "El libro de " + this[section][libro].author + " se llama " + this[section][libro].title;
        }

        libreriaVal = new library("valencia", 0);
        libreriaVal.addSeccion(["aventura", "ciencia ficción", "romántica"]);
        libreriaVal.addSeccion("terror");
        libreriaVal.addBook("ciencia ficción", "La vita è bella", "Roberto Benigni");
