// The most common way of defining custom types is to combine the constructor and prototype patterns.
        //--> Combination Constructor/Prototype Pattern
        function Person(name, age, job) {
            this.name = name;
            this.age = age;
            this.job = job;
        }
        Person.prototype = {
            constructor: Person,
            sayName: function () {
                alert(this.name);
            }
        };
        var peter = new Person("peter", 30, "it staff");
         // with the
         //      constructor: Person
         // we asure that
        console.log(peter instanceof Object); //true
        console.log(peter instanceof Person); // true
        console.log(peter.constructor == Person); //true, without the "constructor:Person" line it would be false
        console.log(peter.constructor == Object); //true
